ÉTAPE 4 — GÉRER LE BACKEND AVEC PM2
═══════════════════════════════════════════════════════════════════════════════

  # 4.1 Installer PM2 globalement (si pas déjà installé)
  npm install -g pm2

  # 4.2 Créer le fichier de configuration PM2
  cat > /opt/portefolia/ecosystem.config.cjs << 'EOF'
  module.exports = {
    apps: [{
      name: 'portefolia-backend',
      script: './backend/src/index.js',
      cwd: '/opt/portefolia',
      instances: 1,
      autorestart: true,
      watch: false,
      max_memory_restart: '512M',
      env: {
        NODE_ENV: 'production',
        PORT: 3001
      },
      error_file: '/var/log/pm2/portefolia-error.log',
      out_file:   '/var/log/pm2/portefolia-out.log',
      log_date_format: 'YYYY-MM-DD HH:mm:ss'
    }]
  };
  EOF

  # 4.3 Créer le dossier de logs
  mkdir -p /var/log/pm2

  # 4.4 Démarrer le backend avec PM2
  cd /opt/portefolia
  pm2 start ecosystem.config.cjs

  # 4.5 Vérifier que le backend tourne
  pm2 status
  pm2 logs portefolia-backend --lines 20

  # 4.6 Enregistrer PM2 pour redémarrage automatique au boot
  pm2 save
  pm2 startup
  # Exécuter la commande affichée par pm2 startup (ex: sudo env PATH=...)

ÉTAPE 6 — CONFIGURER NGINX
═══════════════════════════════════════════════════════════════════════════════

  # 6.1 Vérifier les sites Nginx existants (pour ne pas les toucher)
  ls /etc/nginx/sites-enabled/
  cat /etc/nginx/nginx.conf | grep include

  # 6.2 Créer le bloc Nginx pour Portefolia
  nano /etc/nginx/sites-available/portefolia

  ─────────────────────────────────────────────────────────────
  # /etc/nginx/sites-available/portefolia

  server {
      listen 80;
      server_name portefolia.tech www.portefolia.tech;

      # Redirection HTTP → HTTPS (activée après Certbot, voir étape 7)
      return 301 https://$host$request_uri;
  }

  server {
      listen 443 ssl http2;
      server_name portefolia.tech www.portefolia.tech;

      # ── SSL (sera rempli automatiquement par Certbot) ──────────────────
      ssl_certificate     /etc/letsencrypt/live/portefolia.tech/fullchain.pem;
      ssl_certificate_key /etc/letsencrypt/live/portefolia.tech/privkey.pem;
      include /etc/letsencrypt/options-ssl-nginx.conf;
      ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem;

      # ── Frontend statique ───────────────────────────────────────────────
      root /opt/portefolia/dist;
      index index.html;

      # SPA : toutes les routes non-API vers index.html
      location / {
          try_files $uri $uri/ /index.html;
      }

      # ── Proxy API vers le backend Node.js ──────────────────────────────
      location /api/ {
          proxy_pass         http://127.0.0.1:3001;
          proxy_http_version 1.1;
          proxy_set_header   Upgrade $http_upgrade;
          proxy_set_header   Connection 'upgrade';
          proxy_set_header   Host $host;
          proxy_set_header   X-Real-IP $remote_addr;
          proxy_set_header   X-Forwarded-For $proxy_add_x_forwarded_for;
          proxy_set_header   X-Forwarded-Proto $scheme;
          proxy_cache_bypass $http_upgrade;
          proxy_read_timeout 120s;
          proxy_connect_timeout 10s;
          # Limite upload pour photos/CV (20 MB)
          client_max_body_size 20M;
      }

      # ── Sécurité ────────────────────────────────────────────────────────
      add_header X-Frame-Options SAMEORIGIN;
      add_header X-Content-Type-Options nosniff;
      add_header Referrer-Policy strict-origin-when-cross-origin;

      # ── Compression ─────────────────────────────────────────────────────
      gzip on;
      gzip_types text/plain text/css application/json application/javascript
                 text/xml application/xml image/svg+xml;

      # ── Logs ────────────────────────────────────────────────────────────
      access_log /var/log/nginx/portefolia.access.log;
      error_log  /var/log/nginx/portefolia.error.log;
  }
  ─────────────────────────────────────────────────────────────

  # 6.3 Activer le site (symlink dans sites-enabled)
  ln -s /etc/nginx/sites-available/portefolia /etc/nginx/sites-enabled/portefolia

  # 6.4 Tester la config Nginx (CRITIQUE : ne pas recharger si erreur)
  nginx -t
  # Attendu : "syntax is ok" et "test is successful"

  # 6.5 AVANT le SSL : tester en HTTP d'abord
  #     Commenter temporairement la ligne "return 301 https://..." dans le bloc
  #     port 80 et mettre à la place :
  #
  #       root /opt/portefolia/dist;
  #       index index.html;
  #       location / { try_files $uri $uri/ /index.html; }
  #       location /api/ { proxy_pass http://127.0.0.1:3001; ... }
  #
  #     Puis :
  nginx -t && systemctl reload nginx

ÉTAPE 7 — CERTIFICAT SSL AVEC LET'S ENCRYPT (CERTBOT)
═══════════════════════════════════════════════════════════════════════════════

  # 7.1 Installer Certbot (si pas présent)
  apt update && apt install -y certbot python3-certbot-nginx

  # 7.2 Obtenir le certificat (le DNS doit déjà pointer vers le VPS)
  certbot --nginx -d portefolia.tech -d www.portefolia.tech \
          --email contact@portefolia.tech \
          --agree-tos \
          --no-eff-email

  # Certbot modifie automatiquement le bloc Nginx pour activer HTTPS
  # et configure la redirection HTTP → HTTPS

  # 7.3 Vérifier le renouvellement automatique
  certbot renew --dry-run

  # 7.4 Recharger Nginx après Certbot
  nginx -t && systemctl reload nginx

ÉTAPE 8 — DNS : CONFIGURER LE DOMAINE CHEZ LWS
═══════════════════════════════════════════════════════════════════════════════

  Dans votre espace client LWS → Domaines → DNS Manager de portefolia.tech :

  Supprimer ou remplacer les enregistrements A existants qui pointent vers
  l'hébergement mutualisé LWS, et créer :

  ┌──────────┬──────┬────────────────────────┬────────────────────────────┐
  │ Nom      │ Type │ Valeur                 │ TTL                        │
  ├──────────┼──────┼────────────────────────┼────────────────────────────┤
  │ @        │ A    │ <IP_DE_VOTRE_VPS>      │ 3600                       │
  │ www      │ A    │ <IP_DE_VOTRE_VPS>      │ 3600  (ou CNAME vers @)    │
  │ mail     │ MX   │ conserver les MX LWS   │ (ne pas toucher les MX)   │
  └──────────┴──────┴────────────────────────┴────────────────────────────┘

  IMPORTANT : Ne touchez PAS aux enregistrements MX, SPF, DKIM — ils sont
  nécessaires pour que vos emails SMTP LWS continuent de fonctionner.

  Récupérez l'IP de votre VPS :
    ip addr show eth0 | grep "inet "  (ou hostname -I)

  La propagation DNS peut prendre 1 à 48h.
  Vérifier avec : nslookup portefolia.tech

  Créer un endpoint : https://portefolia.tech/api/stripe/webhook

    - checkout.session.completed
    - customer.subscription.updated
    - customer.subscription.deleted
    - invoice.payment_succeeded
    - invoice.payment_failed

    STRIPE_WEBHOOK_SECRET=whsec_xxxxxxxxxxxxx

  Puis redémarrer le backend :
    pm2 restart portefolia-backend
